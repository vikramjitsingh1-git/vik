# Linoop AWS CICD Pipeline Code Deployment to AWS EC2 Instance


#<b>User Data for Dependencies installations for AMAZON Linux:-</b>

#!/bin/bash<br />
sudo yum -y update<br />
sudo yum -y install ruby<br />
sudo yum -y install wget<br />
cd /home/ec2-user<br />
wget https://aws-codedeploy-us-east-1.s3.us-east-1.amazonaws.com/latest/install<br />
sudo chmod +x ./install<br />
sudo ./install auto<br />
sudo yum install -y python-pip<br />
sudo pip install awscli<br />
